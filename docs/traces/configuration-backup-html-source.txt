            .fetch({method: "GET", data: {numDaysBefore: 1, numDaysAfter: 14}})
            .then((resp) => {
                if (resp?.data?.length > 0) {
                    const maintenanceMessages = resp.data
                        .filter((entry) => entry.data.category === 'Maintenance')
                        .filter((entry) => entry.data.status === 'Scheduled')
                        .sort((l, r) => l.data.startTime - r.data.startTime);
                    if (maintenanceMessages.length > 0 && localStorage.getItem('hideNotificationMessages') !== 'true') {
                        const lastMaintenanceMessage = maintenanceMessages[0];
                        // const lastMaintenanceMessage = maintenanceMessages[maintenanceMessages.length - 1];

                            `Maintenance scheduled from ${Core.StringDateFromMs(lastMaintenanceMessage.data.startTime)} to ${Core.StringDateFromMs(lastMaintenanceMessage.data.endTime)}`
                    }
                }
            });
